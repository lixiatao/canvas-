* 移动画笔，使画笔移动至绘画的开始处(moveTo())
		* 确定第一笔的停止点(lineTo())
		* 规划好之后，选择画笔（包括画笔的粗细和颜色等）(lineWidth、strokeStyle、lineCap、lineJoin...)
		* 确定绘制(fill()、stroke())
		var canvas = document.getElementById('canvas');
		var ctx = canvas.getContext('2d');
		// 横向线
		ctx.beginPath();
		ctx.moveTo(50,50);
		ctx.lineTo(150,50);
		ctx.lineWidth = 8;
		ctx.lineCap = 'square';
		ctx.stroke();
		// 竖向线
		ctx.beginPath();
		ctx.moveTo(50,150);
		ctx.lineTo(50,50);
		ctx.lineCap = 'round';
		ctx.lineWidth = 20;
		ctx.strokeStyle = '#ff00ff';
		ctx.stroke();
		
		// 正方形
		ctx.beginPath();
		ctx.moveTo(250,50);
		ctx.lineTo(450,50);
		ctx.lineTo(450,250);
		ctx.lineTo(250,250);
		ctx.lineTo(250,50);
		ctx.closePath();
		ctx.fillStyle = 'pink';
		ctx.lineWidth = 20;
		ctx.strokeStyle = '#000';
		ctx.lineJoin = 'round'
		ctx.fill();
		ctx.stroke();
		
		// 三角形
		ctx.beginPath();
		ctx.moveTo(500,50);
		ctx.lineTo(550,100);
		ctx.lineTo(600,50);
		ctx.strokeStyle = '#00ff00';
		ctx.closePath();
		ctx.lineJoin = 'round'
		ctx.fill();
		// 五角形
